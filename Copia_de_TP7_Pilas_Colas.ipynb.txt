        "# Ejercicio 1: Pila con nodos\n",
        "class Nodo:\n",
        "    def __init__(self, dato):\n",
        "        self.dato = dato\n",
        "        self.siguiente = None\n",
        "\n",
        "class Pila:\n",
        "    def __init__(self):\n",
        "        self.cima = None\n",
        "\n",
        "    def es_vacia(self):\n",
        "        return self.cima is None\n",
        "\n",
        "    def push(self, dato):\n",
        "        nuevo_nodo = Nodo(dato)\n",
        "        nuevo_nodo.siguiente = self.cima\n",
        "        self.cima = nuevo_nodo\n",
        "\n",
        "    def pop(self):\n",
        "        if self.es_vacia():\n",
        "            return None\n",
        "        valor = self.cima.dato\n",
        "        self.cima = self.cima.siguiente\n",
        "        return valor\n",
        "\n",
        "    def tope(self):\n",
        "        if self.es_vacia():\n",
        "            return None\n",
        "        return self.cima.dato\n",
        "\n",
        "    def __str__(self):\n",
        "        actual = self.cima\n",
        "        elementos = []\n",
        "        while actual:\n",
        "            elementos.append(str(actual.dato))\n",
        "            actual = actual.siguiente\n",
        "        return \" -> \".join(elementos) if elementos else \"Pila vacía\"\n",
        "pila = Pila()\n",
        "pila.push(5)\n",
        "pila.push(3)\n",

        "class Nodo:\n",
        "    def __init__(self, dato, prioridad):\n",
        "        self.dato = dato\n",
        "        self.prioridad = prioridad\n",
        "        self.siguiente = None\n",
        "\n",
        "class ColaPrioridad:\n",
        "    def __init__(self):\n",
        "        self.frente = None\n",
        "\n",
        "    def enqueue(self, dato, prioridad):\n",
        "        nuevo = Nodo(dato, prioridad)\n",
        "        if not self.frente or prioridad < self.frente.prioridad:\n",
        "            nuevo.siguiente = self.frente\n",
        "            self.frente = nuevo\n",
        "        else:\n",
        "            actual = self.frente\n",
        "            while actual.siguiente and actual.siguiente.prioridad <= prioridad:\n",
